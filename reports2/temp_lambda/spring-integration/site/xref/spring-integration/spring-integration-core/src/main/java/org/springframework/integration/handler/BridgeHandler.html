17  package org.springframework.integration.handler;
18  
19  import org.springframework.integration.IntegrationPatternType;
20  import org.springframework.messaging.Message;
21  
22  /**
23   * A simple MessageHandler implementation that passes the request Message
24   * directly to the output channel without modifying it. The main purpose of this
25   * handler is to bridge a PollableChannel to a SubscribableChannel or
26   * vice-versa.
27   * <p>
28   * The BridgeHandler can be used as a stopper at the end of an assembly line of
29   * channels. In this setup the output channel doesn't have to be set, but if the
30   * output channel is omitted the <tt>REPLY_CHANNEL</tt> MUST be set on the
31   * message. Otherwise, a MessagingException will be thrown at runtime.
32   *
33   * @author Mark Fisher
34   * @author Iwein Fuld
35   * @author Marius Bogoevici
36   */
37  public class BridgeHandler extends AbstractReplyProducingMessageHandler {
38  
39  	@Override
40  	public String getComponentType() {
41  		return "bridge";
42  	}
43  
44  	@Override
45  	public IntegrationPatternType getIntegrationPatternType() {
46  		return IntegrationPatternType.bridge;
47  	}
48  
49  	@Override
50  	protected Object handleRequestMessage(Message<?> requestMessage) {
51  		return requestMessage;
52  	}
53  
54  	@Override
55  	protected boolean shouldCopyRequestHeaders() {
56  		return false;
57  	}
58  }
