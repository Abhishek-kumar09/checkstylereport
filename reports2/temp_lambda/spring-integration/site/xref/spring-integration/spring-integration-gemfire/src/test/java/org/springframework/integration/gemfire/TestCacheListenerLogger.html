17  package org.springframework.integration.gemfire;
18  
19  import org.apache.commons.logging.Log;
20  import org.apache.commons.logging.LogFactory;
21  import org.apache.geode.cache.EntryEvent;
22  import org.apache.geode.cache.util.CacheListenerAdapter;
23  
24  /**
25   * (this is the CacheLogger class that ships in the Spring-Gemfire samples)
26   *
27   * @author Costin Leau
28   */
29  public class TestCacheListenerLogger extends CacheListenerAdapter<Object, Object> {
30  
31  	private static final Log log = LogFactory.getLog(TestCacheListenerLogger.class);
32  
33  	@Override
34  	public void afterCreate(EntryEvent<Object, Object> event) {
35  		log.info("Added " + messageLog(event) + " to the cache");
36  	}
37  
38  	@Override
39  	public void afterDestroy(EntryEvent<Object, Object> event) {
40  		log.info("Removed " + messageLog(event) + " from the cache");
41  	}
42  
43  	@Override
44  	public void afterUpdate(EntryEvent<Object, Object> event) {
45  		log.info("Updated " + messageLog(event) + " in the cache");
46  	}
47  
48  	private String messageLog(EntryEvent<Object, Object> event) {
49  		Object key = event.getKey();
50  		Object value = event.getNewValue();
51  
52  		if (event.getOperation().isUpdate()) {
53  			return "[" + key + "] from [" + event.getOldValue() + "] to [" + event.getNewValue() + "]";
54  		}
55  		return "[" + key + "=" + value + "]";
56  	}
57  }
